20 Warehouse Edit Command



-------20.1 Generate Warehouse Locations

DECLARE @WarehouseID int = [TF_9916_WarehouseID]
DECLARE @CountAisle int = (select count(code) from UDF_WM_WarehouseAisle WHERE UDF_WM_WarehouseAisle.WarehouseID = @WarehouseID)
DECLARE @CountRack int = (select count(code) from UDF_WM_WarehouseRack WHERE UDF_WM_WarehouseRack.WarehouseID = @WarehouseID)
DECLARE @CountShell int = (select count(code) from UDF_WM_WarehouseShell WHERE UDF_WM_WarehouseShell.WarehouseID = @WarehouseID)
DECLARE @CountPallet int = (select count(code) from UDF_WM_WarehousePallet WHERE UDF_WM_WarehousePallet.WarehouseID = @WarehouseID)



IF (@CountAisle = 0)
BEGIN
	INSERT INTO UDF_WM_WarehouseAisle(WarehouseID, CODE) Values (@WarehouseID, 'A1')
END

IF (@CountRack = 0)
BEGIN
	INSERT INTO UDF_WM_WarehouseRack(WarehouseID, CODE) Values (@WarehouseID, '')
END

IF (@CountShell = 0)
BEGIN
	INSERT INTO UDF_WM_WarehouseShell(WarehouseID, CODE) Values (@WarehouseID, '')
END

IF (@CountPallet = 0)
BEGIN
	INSERT INTO UDF_WM_WarehousePallet(WarehouseID, CODE) Values (@WarehouseID, '')
END


INSERT INTO [dbo].[AppWarehouseLocation]
           ([CieWarehouseID]
           ,[LocationCode]
           ,[AisleNumber]
           ,[RackNumber]
           ,[ShellNumber]
           ,[PalletNumber]
)
SELECT
	@WarehouseID as CieWarehouseID,
	UDF_WM_WarehouseAisle.Code + UDF_WM_WarehouseRack.Code + UDF_WM_WarehouseShell.Code + UDF_WM_WarehousePallet.Code as LocationCode,
	UDF_WM_WarehouseAisle.AisleID, 
	UDF_WM_WarehouseRack.RackID,
	UDF_WM_WarehouseShell.ShellID,
	UDF_WM_WarehousePallet.PalletID
FROM            
	(select * from UDF_WM_WarehouseAisle WHERE UDF_WM_WarehouseAisle.WarehouseID = @WarehouseID) as UDF_WM_WarehouseAisle
	CROSS JOIN
	(select * from UDF_WM_WarehouseRack WHERE UDF_WM_WarehouseRack.WarehouseID = @WarehouseID) as UDF_WM_WarehouseRack
	CROSS JOIN
	(select * from UDF_WM_WarehouseShell WHERE UDF_WM_WarehouseShell.WarehouseID = @WarehouseID) as UDF_WM_WarehouseShell
	CROSS JOIN
	(select * from UDF_WM_WarehousePallet WHERE UDF_WM_WarehousePallet.WarehouseID = @WarehouseID) as UDF_WM_WarehousePallet 
--WHERE 
ORDER BY UDF_WM_WarehouseAisle.AisleID, 
	UDF_WM_WarehouseRack.RackID,
	UDF_WM_WarehouseShell.ShellID,
	UDF_WM_WarehousePallet.PalletID
